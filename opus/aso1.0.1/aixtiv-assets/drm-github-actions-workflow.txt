name: Dr. Memoria Content Engine CI/CD

on:
  push:
    branches: 
      - main
      - develop
  pull_request:
    branches:
      - main
      - develop

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v3
      with:
        python-version: '3.10'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
        pip install -r requirements-dev.txt
    
    - name: Run unit tests
      run: |
        python -m pytest tests/unit
    
    - name: Run integration tests
      run: |
        python -m pytest tests/integration
    
    - name: Security scanning
      run: |
        pip install bandit
        bandit -r .

  build-and-push:
    needs: test
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write

    steps:
    - uses: actions/checkout@v3
    
    - name: Log in to GitHub Container Registry
      uses: docker/login-action@v2
      with:
        registry: ${{ env.REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}
    
    - name: Extract metadata for Docker
      id: meta
      uses: docker/metadata-action@v4
      with:
        images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
        tags: |
          type=ref,event=branch
          type=ref,event=pr
          type=semver,pattern={{version}}
          type=sha
    
    - name: Build and push Docker image
      uses: docker/build-push-action@v3
      with:
        context: .
        push: ${{ github.event_name != 'pull_request' }}
        tags: ${{ steps.meta.outputs.tags }}
        labels: ${{ steps.meta.outputs.labels }}
        target: production

  deploy-staging:
    needs: build-and-push
    if: github.ref == 'refs/heads/develop'
    runs-on: ubuntu-latest
    environment: staging

    steps:
    - uses: actions/checkout@v3

    - name: Configure Kubernetes credentials
      uses: azure/k8s-set-context@v2
      with:
        method: service-account
        k8s-url: ${{ secrets.KUBE_SERVER }}
        k8s-secret: ${{ secrets.KUBE_TOKEN }}

    - name: Deploy to Staging Kubernetes Cluster
      run: |
        # Deploy to staging namespace
        kubectl apply -f k8s/staging/
        
        # Update deployment image
        kubectl set image deployment/content-engine-deployment \
          content-engine=${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
        
        # Verify deployment
        kubectl rollout status deployment/content-engine-deployment

  deploy-production:
    needs: build-and-push
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    environment: production
    
    steps:
    - uses: actions/checkout@v3

    - name: Configure Kubernetes credentials
      uses: azure/k8s-set-context@v2
      with:
        method: service-account
        k8s-url: ${{ secrets.PROD_KUBE_SERVER }}
        k8s-secret: ${{ secrets.PROD_KUBE_TOKEN }}

    - name: Deploy to Production Kubernetes Cluster
      run: |
        # Apply production manifests
        kubectl apply -f k8s/production/
        
        # Update deployment image
        kubectl set image deployment/content-engine-deployment \
          content-engine=${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
        
        # Verify deployment
        kubectl rollout status deployment/content-engine-deployment
        
        # Optional: Run post-deployment health checks
        ./scripts/health-check.sh

  notify:
    needs: [test, build-and-push, deploy-staging, deploy-production]
    if: always()
    runs-on: ubuntu-latest
    
    steps:
    - name: Send Slack Notification
      uses: 8398a7/action-slack@v3
      with:
        status: ${{ job.status }}
        fields: repo,message,commit,author,action,eventName,ref,workflow
      env:
        SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK }}
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
